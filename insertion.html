#include <iostream>
#include <Windows.h>
#include <vector>

using namespace std;

int main()
{
    system("title insertion sort");
    int input,tmp;
    vector <int> v;
    for(;;){
        cin>>input;
        if(input==0) break;
        v.push_back(input);
    }
    for(int i=0;i<v.size();i++){
        for(int a=i; a!=0; a--){
            if(v[a]<v[a-1]){
                tmp=v[a];
                v[a]=v[a-1];
                v[a-1]=tmp;
            }else{
                break;
            }
        }
    }
    for(int i=0;i<v.size();i++){
        cout<<v[i]<<" ";
    }

    return 0;
}
