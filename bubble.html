#include <iostream>
#include <Windows.h>

using namespace std;

int main(){
  int valtozas;
  int input;
  int tmp;
  int arry[6];
  for(int i=0;i<6;i++){
       	cin>>input;
       	arry[i]=input;
  }
  do{
       	valtozas=0;
       	for(int i=0;i<5;i++){
          	if(arry[i]>arry[i+1]){
               	tmp=arry[i];
               	arry[i]=arry[i+1];
               	arry[i+1]=tmp;
               	valtozas=1;
        }
  }
  }while(valtozas!=0);
  for(int i=0; i<6; i++){
        cout<<arry[i]<<endl;
  }
  return 0;
}
